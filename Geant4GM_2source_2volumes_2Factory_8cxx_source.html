187//_____________________________________________________________________________
188VGM::ISolid* Geant4GM::Factory::ImportSolid(G4VSolid* solid)
189{
191
192 // Do not import the same solid twice
193 //
194 VGM::ISolid* importedSolid = Geant4GM::SolidMap::Instance()->GetSolid(solid);
195 if (importedSolid) return importedSolid;
196
197 if (Debug() > 0) {
198 BaseVGM::DebugInfo();
199 std::cout << "Importing solid: ";
200 if (Debug() > 1) std::cout << solid << " ";
201 std::cout << solid->GetName() << std::endl;
202 }
203
204 G4VSolid* consSolid = solid;
205
206 // Get constituent solid if the solid is a reflected solid
207 //
208 G4ReflectedSolid* reflSolid = dynamic_cast<G4ReflectedSolid*>(solid);
209 if (reflSolid) consSolid = reflSolid->GetConstituentMovedSolid();
210
211 G4Box* box = dynamic_cast<G4Box*>(consSolid);
212 if (box) {
213 return Register(new Geant4GM::Box(box, reflSolid));
214 }
215
216 G4Cons* cons = dynamic_cast<G4Cons*>(consSolid);
217 if (cons) {
218 return Register(new Geant4GM::Cons(cons, reflSolid));
219 }
220
221 G4CutTubs* ctubs = dynamic_cast<G4CutTubs*>(consSolid);
222 if (ctubs) {
223 return Register(new Geant4GM::Ctubs(ctubs, reflSolid));
224 }
225
226 G4Ellipsoid* ellipsoid = dynamic_cast<G4Ellipsoid*>(consSolid);
227 if (ellipsoid) {
228 return Register(new Geant4GM::Ellipsoid(ellipsoid, reflSolid));
229 }
230
231 G4EllipticalTube* eltu = dynamic_cast<G4EllipticalTube*>(consSolid);
232 if (eltu) {
233 return Register(new Geant4GM::EllipticalTube(eltu, reflSolid));
234 }
235
236 G4ExtrudedSolid* xtru = dynamic_cast<G4ExtrudedSolid*>(consSolid);
237 if (xtru) {
238 return Register(new Geant4GM::ExtrudedSolid(xtru, reflSolid));
239 }
240
241 G4Hype* hype = dynamic_cast<G4Hype*>(consSolid);
242 if (hype) {
243 return Register(new Geant4GM::Hype(hype, reflSolid));
244 }
245
246 G4Para* para = dynamic_cast<G4Para*>(consSolid);
247 if (para) {
248 return Register(new Geant4GM::Para(para, reflSolid));
249 }
250
251 G4Paraboloid* paraboloid = dynamic_cast<G4Paraboloid*>(consSolid);
252 if (paraboloid) {
253 return Register(new Geant4GM::Paraboloid(paraboloid, reflSolid));
254 }
255
256 G4Polycone* polycone = dynamic_cast<G4Polycone*>(consSolid);
257 if (polycone) {
258 return Register(new Geant4GM::Polycone(polycone, reflSolid));
259 }
260
261 G4Polyhedra* polyhedra = dynamic_cast<G4Polyhedra*>(consSolid);
262 if (polyhedra) {
263 return Register(new Geant4GM::Polyhedra(polyhedra, reflSolid));
264 }
265
266 G4Sphere* sphere = dynamic_cast<G4Sphere*>(consSolid);
267 if (sphere) {
268 return Register(new Geant4GM::Sphere(sphere, reflSolid));
269 }
270
271 G4TessellatedSolid* tessel = dynamic_cast<G4TessellatedSolid*>(consSolid);
272 if (tessel) {
273 return Register(new Geant4GM::TessellatedSolid(tessel, reflSolid));
274 }
275
276 G4Torus* torus = dynamic_cast<G4Torus*>(consSolid);
277 if (torus) {
278 return Register(new Geant4GM::Torus(torus, reflSolid));
279 }
280
281 G4Trap* trap = dynamic_cast<G4Trap*>(consSolid);
282 if (trap) {
283 return Register(new Geant4GM::Trap(trap, reflSolid));
284 }
285
286 G4Trd* trd = dynamic_cast<G4Trd*>(consSolid);
287 if (trd) {
288 return Register(new Geant4GM::Trd(trd, reflSolid));
289 }
290
291 G4Tubs* tubs = dynamic_cast<G4Tubs*>(consSolid);
292 if (tubs) {
293 return Register(new Geant4GM::Tubs(tubs, reflSolid));
294 }
295
296 G4DisplacedSolid* displaced = dynamic_cast<G4DisplacedSolid*>(consSolid);
297 if (displaced) {
298 ImportSolid(displaced->GetConstituentMovedSolid());
299 return Register(new Geant4GM::DisplacedSolid(displaced, reflSolid));
300 }
301
302 G4ScaledSolid* scaled = dynamic_cast<G4ScaledSolid*>(consSolid);
303 if (scaled) {
304 ImportSolid(scaled->GetUnscaledSolid());
305 return Register(new Geant4GM::ScaledSolid(scaled, reflSolid));
306 }
307
308 G4BooleanSolid* boolean = dynamic_cast<G4BooleanSolid*>(consSolid);
309 if (boolean) {
310 ImportConstituentSolid(0, boolean);
311 ImportConstituentSolid(1, boolean);
312 VGM::IBooleanSolid* vgmBoolean =
313 new Geant4GM::BooleanSolid(boolean, reflSolid);
314 Register(vgmBoolean);
315
316 if (Debug() > 0) {
317 BaseVGM::DebugInfo();
318 std::cout << "Imported Boolean solid: ";
319 if (Debug() > 1) std::cout << vgmBoolean;
320 std::cout << std::endl;
321 BaseVGM::DebugInfo();
322 std::cout << *vgmBoolean << std::endl;
323 }
324
325 return vgmBoolean;
326 }
327
328 G4MultiUnion* multiUnion = dynamic_cast<G4MultiUnion*>(consSolid);
329 if (multiUnion) {
330 for (G4int i = 0; i < multiUnion->GetNumberOfSolids(); ++i) {
331 ImportConstituentSolid(i, multiUnion);
332 }
333 VGM::IMultiUnion* vgmMultiUnion =
334 new Geant4GM::MultiUnion(multiUnion, reflSolid);
335 Register(vgmMultiUnion);
336
337 if (Debug() > 0) {
338 BaseVGM::DebugInfo();
339 std::cout << "Imported MultiUnion solid: ";
340 if (Debug() > 1) std::cout << vgmMultiUnion;
341 std::cout << std::endl;
342 BaseVGM::DebugInfo();
343 std::cout << *vgmMultiUnion << std::endl;
344 }
345
346 return vgmMultiUnion;
347 }
348
349 std::cerr << "Geant4GM::Factory::ImportSolid: " << std::endl;
350 std::cerr << "Unsupported solid type (solid \"" << solid->GetName() << "\""
351 << " type \"" << solid->GetEntityType() << "\")" << std::endl;
352
353 if (Ignore()) {
354 std::cerr << "*** Warning: Using a box instead ***" << std::endl;
355 return Register(new Geant4GM::Box(solid->GetName(),
356 DummyBoxDimensions() / ClhepVGM::Units::Length(),
357 DummyBoxDimensions() / ClhepVGM::Units::Length(),
358 DummyBoxDimensions() / ClhepVGM::Units::Length()));
359 }
360 else {
361 std::cerr << "*** Error: Aborting execution ***" << std::endl;
362 exit(1);
363 }
364}

416void Geant4GM::Factory::ImportDaughters(G4LogicalVolume* lv)
417{
418 // Import recursively all daughters logical volumes.
419
420 if (Debug()) {
421 BaseVGM::DebugInfo();
422 std::cout << "ImportDaughters for " << lv->GetName() << std::endl;
423 }
424
425 for (size_t i = 0; i < lv->GetNoDaughters(); i++) {
426
427 G4LogicalVolume* dLV = lv->GetDaughter(i)->GetLogicalVolume();
428
429 VGM::IVolume* dVolume = Geant4GM::VolumeMap::Instance()->GetVolume(dLV);
430
431 if (!dVolume) {
432 // Import logical volume
433 dVolume = ImportLV(dLV);
434
435 // Process its daughters
436 ImportDaughters(dLV);
437 }
438 }

554VGM::IPlacement* Geant4GM::Factory::ImportPVPair(VGM::IVolume* volume,
555 VGM::IVolume* motherVolume, G4PhysicalVolumesPair pvPair)
556{
557 G4ReflectionFactory* g4ReflectionFactory = G4ReflectionFactory::Instance();
558
559 // Get info about created PV
560 G4VPhysicalVolume* pv1 = pvPair.first;
561 G4LogicalVolume* lv1 = pv1->GetLogicalVolume();
562 VGM::IVolume* iv1 = Geant4GM::VolumeMap::Instance()->GetVolume(lv1);
563 if (!iv1) {
564 // a new logical volume has been created by G4 reflection factory
565 // has to be imported into VGM
566 iv1 = ImportLV(lv1, volume->MediumName());
567 ImportDaughters(lv1);
568 ImportPositions(lv1);
569 }
570 VGM::IPlacement* placement1 = new Geant4GM::Placement(iv1, motherVolume, pv1);
571 // Register physical volume in the map
572 Geant4GM::PlacementMap::Instance()->AddPlacement(placement1, pv1);
573
574 G4VPhysicalVolume* pv2 = pvPair.second;
575 if (pv2) {
576 G4LogicalVolume* g4MotherLV =
577 Geant4GM::VolumeMap::Instance()->GetVolume(motherVolume);
578
579 // mother volume is the other volume in the pair
580 // constituen/reflected mother lv
581 G4LogicalVolume* mlv2 = 0;
582 if (g4ReflectionFactory->IsReflected(g4MotherLV))
583 mlv2 = g4ReflectionFactory->GetConstituentLV(g4MotherLV);
584 if (g4ReflectionFactory->IsConstituent(g4MotherLV))
585 mlv2 = g4ReflectionFactory->GetReflectedLV(g4MotherLV);
586 if (!mlv2) {
587 // should not happen
588 std::cerr << " Geant4GM::Factory::CreatePlacement: " << std::endl;
589 std::cerr << " Misundersood G4ReflectionFactory behavior "
590 << std::endl;
591 std::cerr << "*** Error: Aborting execution ***" << std::endl;
592 exit(1);
593 }
594 else {
595 VGM::IVolume* miv2 = Geant4GM::VolumeMap::Instance()->GetVolume(mlv2);
596 if (!miv2) {
597 // should not happen
598 std::cerr << " Geant4GM::Factory::CreatePlacement: " << std::endl;
599 std::cerr << " Missing mapping of existing LV to VGM" << std::endl;
600 std::cerr << "*** Error: Aborting execution ***" << std::endl;
601 exit(1);
602 }
603
604 G4LogicalVolume* lv2 = pv2->GetLogicalVolume();
605 VGM::IVolume* iv2 = Geant4GM::VolumeMap::Instance()->GetVolume(lv2);
606 if (!iv2) {
607 // a new logical volume has been created by G4 reflection factory
608 // has to be imported into VGM
609 iv2 = ImportLV(lv2, volume->MediumName());
610 ImportDaughters(lv2);
611 ImportPositions(lv2);
612 }
613 VGM::IPlacement* placement2 = new Geant4GM::Placement(iv2, miv2, pv2);
614 // Register physical volume in the map
615 Geant4GM::PlacementMap::Instance()->AddPlacement(placement2, pv2);
616 }
617 }
618
619 // Register physical volume in the map
620 Geant4GM::PlacementMap::Instance()->AddPlacement(placement1, pv1);
621
622 return placement1;
623 // should allow to return a list of placements
624}

629{
635

642

644 }
645

650 VGM::ISolid* vgmSolid = new Geant4GM::Polycone(tubs);
651 dynamic_cast<Geant4GM::Volume*>(volume)->ResetSolid(vgmSolid);
652
653 return true;
654 }
655
656 return false;

1060VGM::IPlacement* Geant4GM::Factory::CreateMultiplePlacement(
1061 const std::string& name, VGM::IVolume* volume, VGM::IVolume* motherVolume,
1062 VGM::Axis axis, int nofItems, double width, double offset, double halfGap)
1063{
1064 //
1065
1066 // Top volume
1067 if (!motherVolume) {
1068 std::cerr << "Geant4GM::Factory::CreateMultiplePlacement: "
1069 << " Mother volume not defined!" << std::endl;
1070 std::cerr << "*** Error: Aborting execution ***" << std::endl;
1071 exit(1);
1072 }
1073
1074 // Get logical volumes from the volumes map
1075 G4LogicalVolume* g4LV = Geant4GM::VolumeMap::Instance()->GetVolume(volume);
1076
1077 G4LogicalVolume* g4MotherLV =
1078 Geant4GM::VolumeMap::Instance()->GetVolume(motherVolume);
1079
1080 // Geant4 requires the same type of solid in both
1081 // volume and mother volume
1082 VGM::SolidType solidType = volume->Solid()->Type();
1083 VGM::SolidType motherSolidType = motherVolume->Solid()->Type();
1084 if (solidType != motherSolidType) {
1085 bool result = SwitchSolid(volume, g4LV, g4MotherLV);
1086 if (!result) {
1087 std::cerr << " Geant4GM::Factory::CreateMultiplePlacement: "
1088 << std::endl
1089 << " Different solid types in volume and mother!"
1090 << std::endl
1091 << " volume: " << volume->Name() << " "
1092 << VGM::SolidTypeName(solidType)
1093 << " mother: " << motherVolume->Name() << " "
1094 << VGM::SolidTypeName(motherSolidType) << std::endl
1095 << "*** Error: Aborting execution ***" << std::endl;
1096 exit(1);
1097 }
1098 }
1099
1100 // Apply units
1101 width /= ClhepVGM::Units::AxisUnit(axis);
1102 offset /= ClhepVGM::Units::AxisUnit(axis);
1103
1104 // Update offset if it goes beyond mother dhi
1105 if (axis == VGM::kPhi && offset + nofItems * width > 2 * CLHEP::pi)
1106 offset = offset - 2 * CLHEP::pi;
1107
1108 // The instance of G4PVDivisionFactory must exist
1109 G4PVDivisionFactory::GetInstance();
1110
1111 // The halfGap is not yet supported by G4ReflectionFactory
1112 G4PhysicalVolumesPair pvPair;
1113 if (halfGap != 0.) {
1114 G4VPhysicalVolume* replicatedSlice =
1115 new G4ReplicatedSlice(name, g4LV, g4MotherLV,
1116 Geant4GM::Placement::GetAxis(axis), nofItems, width, halfGap, offset);
1117 pvPair.first = replicatedSlice;
1118 pvPair.second = 0;
1119 }
1120 else {
1121 // Create PV division
1122 // for a general transformation we have to use G4 reflection factory
1123 G4ReflectionFactory* g4ReflectionFactory = G4ReflectionFactory::Instance();
1124
1125 // G4PhysicalVolumesPair pvPair
1126 pvPair = g4ReflectionFactory->Divide(name, g4LV, g4MotherLV,
1127 Geant4GM::Placement::GetAxis(axis), nofItems, width, offset);
1128 }
1129
1130 // Import volumes created via G4 reflection factory
1131 VGM::IPlacement* placement1 = ImportPVPair(volume, motherVolume, pvPair);
1132
1133 return placement1;
1134 // should allow to return a list of placements
1135}
